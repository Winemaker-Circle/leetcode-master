class Solution:
    def titleToNumber(self, columnTitle: str) -> int:
        length = len(columnTitle)
        result = 0
        for i in range(1, length + 1):
            # A=65 A=1
            result += (ord(columnTitle[-i]) - 65 + 1) * math.pow(26, i - 1)

        return int(result)
